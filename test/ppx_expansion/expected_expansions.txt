===== a_nullary.ml =====
type err = [ `Unavailable ][@@deriving eta_error]
include
  struct
    let _ = fun (_ : err) -> ()
    let pp_err : Format.formatter -> err -> unit =
      fun fmt ->
        function | `Unavailable -> Format.pp_print_string fmt "unavailable"
    let _ = pp_err
  end[@@ocaml.doc "@inline"][@@merlin.hide ]

===== h_override.ml =====
type payload = {
  id: string }
let pp_payload fmt payload = Format.pp_print_string fmt payload.id
type err = [ `Custom of payload [@eta.render pp_payload]][@@deriving
                                                           eta_error]
include
  struct
    let _ = fun (_ : err) -> ()
    let pp_err : Format.formatter -> err -> unit =
      fun fmt ->
        function
        | `Custom value -> Format.fprintf fmt "custom:%a" pp_payload value
    let _ = pp_err
  end[@@ocaml.doc "@inline"][@@merlin.hide ]
